class DebugPrint {
  static void print(String texto) {
    if (releaseVersion == false) {
      developer.log(texto.toLowerCase(), name: 'Debug:');
    }
  }

  static void error(String error) {
    if (releaseVersion == false) {
      String _error = "\n##########\n";
      _error += "${error.toUpperCase()}\n";
      _error += "##########\n";
      developer.log("erro", name: 'Error:', error: "$_error");
    }
  }
}
